flop.js is entirely front-end Javascript. Simply copy the Javascript files from flop-vm and flop-render/dist/flop-render.js into subfolders on your web site, and `import` them from there, for example: 

    // import flop.js
    import { RenderLoop } from "./flop-render/flop-render.js";
    import { Sprite, Flop } from "./flop-vm/index.js";
